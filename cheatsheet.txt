php artisan cache:clear

php artisan make:model MyModel

# Generate a new migration
php artisan make:migration create_table_name

# Generate a new seeder
php artisan make:seeder MySeeder

# Generate a new factory
php artisan make:factory MyFactory

# Generate a new middleware
php artisan make:middleware MyMiddleware

# Generate a new provider
php artisan make:provider MyProvider

# Generate a new rule
php artisan make:rule MyRule

# Generate a new command
php artisan make:command MyCommand

# Generate a new event listener
php artisan make:listener MyListener --event=MyEvent

# Generate a new model with migration, factory, and resource
php artisan make:model MyModel --migration --factory --resource
